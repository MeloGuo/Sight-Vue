<template>
  <button class="s-button" :class="`icon-${iconPosition}`" @click="$emit('click')">
    <s-icon class="icon" v-if="icon && !loading" :name="icon"></s-icon>
    <s-icon class="icon loading" v-if="loading" name="loading"></s-icon>
    <div class="content">
      <slot></slot>
    </div>
  </button>
</template>
 
<script>
  import Icon from '../icon/icon.vue'
 
  export default {
    name: 'SightButton',
    components: {
      's-icon': Icon
    },
    props: {
      icon: String,
      loading: {
        type: Boolean,
        default: false
      },
      iconPosition: {
        type: String,
        default: 'left',
        validator (direction) {
          return ['left', 'right'].indexOf(direction) > -1
        }
      }
    }
  }
</script>
